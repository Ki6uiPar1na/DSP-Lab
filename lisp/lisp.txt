(defun demonstrate-list-manipulation ()
  ;; Let's initialize a couple of starting lists
  (let ((numbers '(1 2 3 4 5))
        (letters '(a b c d)))
    
    (format t "--- Initial Data ---~%")
    (format t "Numbers List: ~A~%" numbers)
    (format t "Letters List: ~A~%~%" letters)

    ;; 1. ACCESSORS (Anatomy)
    (format t "--- 1. Accessing Elements ---~%")
    (format t "CAR (First element of letters): ~A~%" (car letters))
    (format t "CDR (Rest of letters): ~A~%" (cdr letters))
    (format t "CADR (Second element using shorthand): ~A~%" (cadr letters))
    (format t "NTH (Element at index 2 of numbers): ~A~%" (nth 2 numbers))
    (format t "NTHCDR (Drop first 2 elements of numbers): ~A~%~%" (nthcdr 2 numbers))

    ;; 2. CONSTRUCTORS (Building Lists)
    (format t "--- 2. Constructing Lists ---~%")
    (format t "CONS (Prepend 'z' to letters): ~A~%" (cons 'z letters))
    (format t "LIST (Combine items into new structure): ~A~%" (list 'x 'y 'z))
    (format t "APPEND (Merge numbers and letters together): ~A~%~%" (append numbers letters))

    ;; 3. STRUCTURAL TRANSFORMERS
    (format t "--- 3. Structural Transformers ---~%")
    (format t "REVERSE (Flip the numbers list): ~A~%" (reverse numbers))
    (format t "LENGTH (Count elements in letters): ~A~%" (length letters))
    (format t "LAST (Get the last cons cell of numbers): ~A~%~%" (last numbers))

    ;; 4. SEARCHING & FILTERING
    (format t "--- 4. Searching & Filtering ---~%")
    (format t "MEMBER (Does 'c' exist in letters?): ~A~%" (member 'c letters))
    (format t "FIND (Find number 4): ~A~%" (find 4 numbers))
    (format t "REMOVE (Strip the number 3 out): ~A~%~%" (remove 3 numbers))

    ;; 5. HIGH-ORDER MAPPING
    (format t "--- 5. High-Order Mapping ---~%")
    (format t "MAPCAR (Square every element in the numbers list): ~A~%" 
            (mapcar #'(lambda (x) (* x x)) numbers))))

;; Run the complete demonstration
(demonstrate-list-manipulation)
